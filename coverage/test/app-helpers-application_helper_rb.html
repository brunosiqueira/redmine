159   def project_tree(projects, &block)
160     ancestors = []
161     projects.sort_by(&:lft).each do |project|
162       while (ancestors.any? && !project.is_descendant_of?(ancestors.last)) 
163         ancestors.pop
164       end
165       yield project, ancestors.size
166       ancestors << project
167     end
168   end
