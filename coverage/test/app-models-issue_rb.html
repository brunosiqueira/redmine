 94   # Returns the moved/copied issue on success, false on failure
 95   def move_to(new_project, new_tracker = nil, options = {})
 96     options ||= {}
 97     issue = options[:copy] ? self.clone : self
 98     transaction do
 99       if new_project && issue.project_id != new_project.id
100         # delete issue relations
101         unless Setting.cross_project_issue_relations?
102           issue.relations_from.clear
103           issue.relations_to.clear
104         end
105         # issue is moved to another project
106         # reassign to the category with same name if any
107         new_category = issue.category.nil? ? nil : new_project.issue_categories.find_by_name(issue.category.name)
108         issue.category = new_category
109         issue.fixed_version = nil
110         issue.project = new_project
111       end
112       if new_tracker
113         issue.tracker = new_tracker
114       end
115       if options[:copy]
116         issue.custom_field_values = self.custom_field_values.inject({}) {|h,v| h[v.custom_field_id] = v.value; h}
117         issue.status = self.status
118       end
119       if issue.save
120         unless options[:copy]
121           # Manually update project_id on related time entries
122           TimeEntry.update_all("project_id = #{new_project.id}", {:issue_id => id})
123         end
124       else
125         Issue.connection.rollback_db_transaction
126         return false
127       end
128     end
129     return issue

348   # Called after_save
349   def create_journal
350     if @current_journal
351       # attributes changes
352       (Issue.column_names - %w(id description lock_version created_on updated_on)).each {|c|
353         @current_journal.details << JournalDetail.new(:property => 'attr',
354                                                       :prop_key => c,
355                                                       :old_value => @issue_before_change.send(c),
356                                                       :value => send(c)) unless send(c)==@issue_before_change.send(c)
357       }
358       # custom fields changes
359       custom_values.each {|c|
360         next if (@custom_values_before_change[c.custom_field_id]==c.value ||
361                   (@custom_values_before_change[c.custom_field_id].blank? && c.value.blank?))
362         @current_journal.details << JournalDetail.new(:property => 'cf', 
363                                                       :prop_key => c.custom_field_id,
364                                                       :old_value => @custom_values_before_change[c.custom_field_id],
365                                                       :value => c.value)
366       }      
367       @current_journal.save
368     end
369   end
