146   def redirect_back_or_default(default)
147     back_url = CGI.unescape(params[:back_url].to_s)
148     if !back_url.blank?
149       begin
150         uri = URI.parse(back_url)
151         # do not redirect user to another host or to the login or register page
152         if (uri.relative? || (uri.host == request.host)) && !uri.path.match(%r{/(login|account/register)})

157       end
158     end

231   # qvalues http header parser
232   # code taken from webrick
233   def parse_qvalues(value)
234     tmp = []
235     if value
236       parts = value.split(/,\s*/)
237       parts.each {|part|
238         if m = %r{^([^\s,]+?)(?:;\s*q=(\d+(?:\.\d+)?))?$}.match(part)
239           val = m[1]
240           q = (m[2] or 1).to_f
241           tmp.push([val, q])
242         end
243       }
244       tmp = tmp.sort_by{|val, q| -q}
245       tmp.collect!{|val, q| val}
246     end
247     return tmp
248   rescue
249     nil
250   end
