269         call_hook(:controller_issues_bulk_edit_before_save, { :params => params, :issue => issue })

306       unsaved_issue_ids = []
307       @issues.each do |issue|
308         issue.init_journal(User.current)
309         unsaved_issue_ids << issue.id unless issue.move_to(@target_project, new_tracker, params[:copy_options])
310       end

316           :ids => '#' + unsaved_issue_ids.join(', #'))

326     if @hours > 0

328       when 'destroy'
329         # nothing to do

335           flash.now[:error] = l(:error_issue_not_found_in_project)

343       end
344     end
345     @issues.each(&:destroy)
