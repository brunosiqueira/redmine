101     # Sets a requirement on Redmine version
102     # Raises a PluginRequirementError exception if the requirement is not met
103     #
104     # Examples
105     #   # Requires Redmine 0.7.3 or higher
106     #   requires_redmine :version_or_higher => '0.7.3'
107     #   requires_redmine '0.7.3'
108     #
109     #   # Requires a specific Redmine version
110     #   requires_redmine :version => '0.7.3'              # 0.7.3 only
111     #   requires_redmine :version => ['0.7.3', '0.8.0']   # 0.7.3 or 0.8.0
112     def requires_redmine(arg)
113       arg = { :version_or_higher => arg } unless arg.is_a?(Hash)
114       arg.assert_valid_keys(:version, :version_or_higher)
115       
116       current = Redmine::VERSION.to_a
117       arg.each do |k, v|
118         v = [] << v unless v.is_a?(Array)
119         versions = v.collect {|s| s.split('.').collect(&:to_i)}
120         case k
121         when :version_or_higher
122           raise ArgumentError.new("wrong number of versions (#{versions.size} for 1)") unless versions.size == 1
123           unless (current <=> versions.first) >= 0
124             raise PluginRequirementError.new("#{id} plugin requires Redmine #{v} or higher but current is #{current.join('.')}")
125           end
126         when :version
127           unless versions.include?(current.slice(0,3))
128             raise PluginRequirementError.new("#{id} plugin requires one the following Redmine versions: #{v.join(', ')} but current is #{current.join('.')}")
129           end
130         end
131       end
132       true
133     end

151     # Defines a permission called +name+ for the given +actions+.
152     # 
153     # The +actions+ argument is a hash with controllers as keys and actions as values (a single value or an array):
154     #   permission :destroy_contacts, { :contacts => :destroy }
155     #   permission :view_contacts, { :contacts => [:index, :show] }
156     # 
157     # The +options+ argument can be used to make the permission public (implicitly given to any user)
158     # or to restrict users the permission can be given to.
159     # 
160     # Examples
161     #   # A permission that is implicitly given to any user
162     #   # This permission won't appear on the Roles & Permissions setup screen
163     #   permission :say_hello, { :example => :say_hello }, :public => true
164     # 
165     #   # A permission that can be given to any user
166     #   permission :say_hello, { :example => :say_hello }
167     #   
168     #   # A permission that can be given to registered users only
169     #   permission :say_hello, { :example => :say_hello }, :require => loggedin
170     #   
171     #   # A permission that can be given to project members only
172     #   permission :say_hello, { :example => :say_hello }, :require => member
173     def permission(name, actions, options = {})
174       if @project_module
175         Redmine::AccessControl.map {|map| map.project_module(@project_module) {|map|map.permission(name, actions, options)}}
176       else
177         Redmine::AccessControl.map {|map| map.permission(name, actions, options)}
178       end
179     end

194     # Registers an activity provider.
195     #
196     # Options:
197     # * <tt>:class_name</tt> - one or more model(s) that provide these events (inferred from event_type by default)
198     # * <tt>:default</tt> - setting this option to false will make the events not displayed by default
199     # 
200     # A model can provide several activity event types.
201     # 
202     # Examples:
203     #   register :news
204     #   register :scrums, :class_name => 'Meeting'
205     #   register :issues, :class_name => ['Issue', 'Journal']
206     # 
207     # Retrieving events:
208     # Associated model(s) must implement the find_events class method.
209     # ActiveRecord models can use acts_as_activity_provider as a way to implement this class method.
210     # 
211     # The following call should return all the scrum events visible by current user that occured in the 5 last days: 
212     #   Meeting.find_events('scrums', User.current, 5.days.ago, Date.today)
213     #   Meeting.find_events('scrums', User.current, 5.days.ago, Date.today, :project => foo) # events for project foo only
214     # 
215     # Note that :view_scrums permission is required to view these events in the activity view.
216     def activity_provider(*args)
