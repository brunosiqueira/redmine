124   def self.allowed_to_condition(user, permission, options={})
125     statements = []
126     base_statement = "#{Project.table_name}.status=#{Project::STATUS_ACTIVE}"
127     if perm = Redmine::AccessControl.permission(permission)
128       unless perm.project_module.nil?
129         # If the permission belongs to a project module, make sure the module is enabled
130         base_statement << " AND EXISTS (SELECT em.id FROM #{EnabledModule.table_name} em WHERE em.name='#{perm.project_module}' AND em.project_id=#{Project.table_name}.id)"
131       end
132     end
133     if options[:project]
134       project_statement = "#{Project.table_name}.id = #{options[:project].id}"
135       project_statement << " OR (#{Project.table_name}.lft > #{options[:project].lft} AND #{Project.table_name}.rgt < #{options[:project].rgt})" if options[:with_subprojects]
136       base_statement = "(#{project_statement}) AND (#{base_statement})"
137     end
138     if user.admin?
139       # no restriction
140     else
141       statements << "1=0"
142       if user.logged?

147         # anonymous user allowed on public project
148         statements << "#{Project.table_name}.is_public = #{connection.quoted_true}"

208   # Argument can be either a Project, a String, a Fixnum or nil
209   def set_parent!(p)
210     unless p.nil? || p.is_a?(Project)
211       if p.to_s.blank?
212         p = nil
213       else
214         p = Project.find_by_id(p)
215         return false unless p
216       end
217     end
218     if p == parent && !p.nil?
219       # Nothing to do
220       true
221     elsif p.nil? || (p.active? && move_possible?(p))
222       # Insert the project so that target's children or root projects stay alphabetically sorted
223       sibs = (p.nil? ? self.class.roots : p.children)
224       to_be_inserted_before = sibs.detect {|c| c.name.to_s.downcase > name.to_s.downcase }
225       if to_be_inserted_before
226         move_to_left_of(to_be_inserted_before)
227       elsif p.nil?
228         if sibs.empty?
229           # move_to_root adds the project in first (ie. left) position
230           move_to_root
231         else
232           move_to_right_of(sibs.last) unless self == sibs.last
233         end
234       else
235         # move_to_child_of adds the project in last (ie.right) position
236         move_to_child_of(p)
237       end
238       true
239     else
240       # Can not move to the given target
241       false
242     end
243   end

340   # Copies and saves the Project instance based on the +project+.
341   # Will duplicate the source project's:
342   # * Issues
343   # * Members
344   # * Queries
345   def copy(project)
346     project = project.is_a?(Project) ? project : Project.find(project)
347 
348     Project.transaction do
349       # Issues
350       project.issues.each do |issue|
351         new_issue = Issue.new
352         new_issue.copy_from(issue)
353         self.issues << new_issue
354       end
355     
356       # Members
357       project.members.each do |member|
358         new_member = Member.new
359         new_member.attributes = member.attributes.dup.except("project_id")
360         new_member.role_ids = member.role_ids.dup
361         new_member.project = self
362         self.members << new_member
363       end
364       
365       # Queries
366       project.queries.each do |query|
367         new_query = Query.new
368         new_query.attributes = query.attributes.dup.except("project_id", "sort_criteria")
369         new_query.sort_criteria = query.sort_criteria if query.sort_criteria
370         new_query.project = self
371         self.queries << new_query
372       end
373 
374       Redmine::Hook.call_hook(:model_project_copy_before_save, :source_project => project, :destination_project => self)
375       self.save
376     end
377   end

380   # Copies +project+ and returns the new instance.  This will not save
381   # the copy
382   def self.copy_from(project)
383     begin
384       project = project.is_a?(Project) ? project : Project.find(project)
385       if project
386         # clear unique attributes
387         attributes = project.attributes.dup.except('name', 'identifier', 'id', 'status')
388         copy = Project.new(attributes)
389         copy.enabled_modules = project.enabled_modules
390         copy.trackers = project.trackers
391         copy.custom_values = project.custom_values.collect {|v| v.clone}
392         return copy
393       else
394         return nil
395       end
396     rescue ActiveRecord::RecordNotFound
397       return nil
398     end
399   end
